// Univ-Bot Main Functionality
function handleStudentRequest(userId, request) {
    // Connect to database
    const db = new SQLite3.Database('univbot.db');
    
    // Check request type
    switch(request.type) {
        case 'course':
            return getCourseMaterials(db, userId, request.course);
        case 'exam':
            return getExamInfo(db, userId, request.exam);
        case 'schedule':
            return getSchedule(db, userId);
        default:
            return "Sorry, I didn't understand that request.";
    }
}

// Example usage
const response = handleStudentRequest('12345', {type: 'course', course: 'CS101'});
console.log('Bot response:', response);
